import java.util.Scanner;
public class quest1{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        System.out.println("Enter the number of test cases that you want to run:");
        int N = sc.nextInt();
        for(int i=1; i<=N; i++){
            
            int X = sc.nextInt();//The first number of the sum series
            
            int Y = sc.nextInt();//The next Yth number im gonna sum

            int count = 0;
            int sum = 0;
            while(count<Y){
                if(X%2!=0){
                sum = sum+X;
                count++;
                }
                X++;
            }
            System.out.println("Sum: "+sum);
        }
    }
}

import java.util.Scanner;
public class quest4{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        System.out.println("Number of Customers:");
        int count = sc.nextInt();
        for(int i=1; i<=count; i++){
            System.out.print("Item number for customer "+i+": ");
            int items = sc.nextInt();
            double sum = 0;
            for(int k=1; k<=items; k++){
            System.out.println("Item_"+k+": ");
            int price = sc.nextInt();
            sum+=price;
            }
            double avg = (double)sum/items;
            System.out.println("Average for customer"+i+"; "+ avg);
        }
    }
}
