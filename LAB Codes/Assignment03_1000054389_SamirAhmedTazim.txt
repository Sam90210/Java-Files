public class quest2{
    public static void main(String[] args){
        int sum =0;
        for(int i=1; i<=600; i++){
            if((i%7==0 || i%9==0) && i%63!=0){/*
             * A number divisible by both 7 and 9 must be
             * divisible by their Least Common Multiple (LCM)
             * and the LCM of 7 and 9 is 7*9 = 63
             Also put the or operation in bracket cause
             accordint to precedence && operator operates before
             than || operator. But bracket holds the top precedence 
             and we need to determine the divsibility by 7 & 9 first
             then cause || operation will work even if the both case
             is true so we gotta check the divisibily by LCM later
             it determines || operation in  brackets!!
             */
            sum = sum+i;
            }
        }
        System.out.println("Summation: "+sum);
    }
}
